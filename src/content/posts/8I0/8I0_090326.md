// unusual mod : 1000003999
// atan2(y, x): tinh goc tao boi oX va diem co toa do (x, y)
struct Skill
{
  ll p; // Vị trí
  ll r; // Lượng hồi phục
};
ll L, S, V1, V2;
ll C1, C2;
int N;
vector<Skill> skills;

bool check(ll start_sprint_pos)
{
  ll current_stamina = S;
  ll current_pos = 0;

  for (int i = 0; i <= N; ++i) {
    ll next_pos = skills[i].p;
    ll dist = next_pos - current_pos;

    // before spurt
    if (next_pos <= start_sprint_pos)
    {
      current_stamina -= dist * C1;
    }
    // after spurt
    else if (current_pos >= start_sprint_pos)
    {
      current_stamina -= dist * C2;
    }
    // inbetween
    else
    {
      ll dist_walk = start_sprint_pos - current_pos;
      ll dist_sprint = next_pos - start_sprint_pos;
      current_stamina -= (dist_walk * C1 + dist_sprint * C2);
    }

    if (current_stamina < 0)
      return false;
    current_stamina += skills[i].r;
    current_pos = next_pos;
  }

  // After last skill
  ll next_pos = L;
  ll dist = next_pos - current_pos;

  if (next_pos <= start_sprint_pos)
    current_stamina -= dist * C1;
  else if (current_pos >= start_sprint_pos)
    current_stamina -= dist * C2;
  else {
    ll dist_walk = start_sprint_pos - current_pos;
    ll dist_sprint = next_pos - start_sprint_pos;
    current_stamina -= (dist_walk * C1 + dist_sprint * C2);
  }

  if (current_stamina < 0)
    return false;

  return true;
}

void Solve()
{
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);

  cin >> L >> S >> V1 >> V2;
  cin >> C1 >> C2;
  cin >> N;

  skills.resize(N + 1);
  for (int i = 0; i < N; ++i)
    cin >> skills[i].p >> skills[i].r;
  // finish line
  skills[N].p = L;
  skills[N].r = 0;
  sort(skills.begin(), skills.end(), [](const Skill &a, const Skill &b){ return a.p < b.p; });
  ll low = 0;
  ll high = L;
  ll ans_pos = -1;

  if (!check(L)){
    cout << -1 << endl;
    return;
  }

  while (low <= high) {
    ll mid = low + (high - low) / 2;
    if (check(mid)) {
      ans_pos = mid;
      high = mid - 1;
    } else {
      low = mid + 1;
    }
  }

  if (ans_pos == -1) {
    cout << -1 << endl;
  } else{
    double time = (double)ans_pos / V1 + (double)(L - ans_pos) / V2;
    cout << fixed << setprecision(6) << time << endl;
  }
}

int32_t main()
{
  using namespace chrono;
  auto start = high_resolution_clock::now();
  FileInOut();

  int test = 1;
  // cin >> test;
  while (test--)
    Solve();

  auto stop = high_resolution_clock::now();
  auto duration = duration_cast<microseconds>(stop - start);
  double t = duration.count() * 1.0 / 1000000;
  cerr << "Runtime: " << t << "s" << endl;
}
