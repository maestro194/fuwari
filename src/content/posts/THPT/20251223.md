# **Bài 1: Nối mạng**

<details>
<summary>Code mẫu</summary>

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 3e5 + 5;
int n, m;
bool vis[MAXN];
vector<int> adj[MAXN];

void bfs(int s){
    queue<int> q;
    q.push(s);
    while(!q.empty()){
        int k=q.front();
        q.pop();
        for(int x : adj[k]){
            if(!vis[x]){
                vis[x]=1;
                q.push(x);
            }
        }
    }
}

signed main() {
    ios_base::sync_with_stdio(0);
    cin.tie(nullptr); cout.tie(nullptr);
    cin >> n >> m;
    int res = 1e9;
    vis[1] = 1;
    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        if (vis[u]) bfs(u);
        if (vis[n]) res = min(res, i);
    }
    cout << (res == 1e9 ? -1 : res);
    return 0;
}
```
