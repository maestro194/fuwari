# **Bài 4: Giá sách**

<details>
<summary>Code mẫu</summary>

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
constexpr int MAXN = 1e5;

int n;

struct TPoint
{
    int x, y;
    bool operator < (const TPoint& other) const
    {
        return x < other.x;
    }
};

struct TShelf
{
    int x1, x2, y;
    bool operator < (const TShelf& other) const
    {
        return y > other.y;
    }
} Shelf[MAXN];

void Enter()
{
    cin >> n;
    for (int i = 0; i < n; ++i)
        cin >> Shelf[i].x1 >> Shelf[i].x2 >> Shelf[i].y;
    sort(Shelf, Shelf + n);
}

void Solve()
{
    long long Result = 0;
    set<TPoint> S;
    for (int i = 0; i < n; ++i) {
        TPoint A = {Shelf[i].x1, Shelf[i].y};
        TPoint B = {Shelf[i].x2, Shelf[i].y};
        auto it = S.lower_bound(A);
        auto it2 = it;
        for (; it2 != S.end() && it2->x <= B.x; ++it2)
            Result += it2->y - A.y;
        S.erase(it, it2);
        S.insert(A); S.insert(B);
    }
    for (const TPoint& p: S) Result += p.y;
    cout << Result;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    Enter(); Solve();
}
```
